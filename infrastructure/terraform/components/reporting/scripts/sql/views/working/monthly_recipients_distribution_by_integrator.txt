WITH monthly_messages AS (
    SELECT
        DATE(DATE_TRUNC('month', createdtime)) AS createdMonth,
        clientid,
        nhsnumberhash,
        COUNT(requestitemid) AS messagesDelivered
    FROM request_item_status
    CROSS JOIN UNNEST(completedcommunicationtypes) AS t(communicationType)
    WHERE status = 'DELIVERED'
    GROUP BY 1, 2, 3
)

SELECT
    createdMonth,
    clientid,
    messagesDelivered AS totalMessagesByIntegrator,
    COUNT(DISTINCT nhsnumberhash) AS countOfRecipients
FROM monthly_messages
GROUP BY 1, 2, 3
