SELECT
    createdMonth,
    monthlyTotalMessages,
    nhsnumberhash,
    clientid,
    SUM(CASE WHEN communicationType = 'NHSAPP' THEN messagesDelivered ELSE 0 END) AS AppDelivered,
    SUM(CASE WHEN communicationType = 'EMAIL' THEN messagesDelivered ELSE 0 END) AS EmailDelivered,
    SUM(CASE WHEN communicationType = 'SMS' THEN messagesDelivered ELSE 0 END) AS SMSDelivered,
    SUM(CASE WHEN communicationType = 'LETTER' THEN messagesDelivered ELSE 0 END) AS LetterDelivered
FROM (
    SELECT
        DATE(DATE_TRUNC('month', createdtime)) AS createdMonth,
        nhsnumberhash,
        communicationType,
        clientid,
        COUNT(requestitemid) AS messagesDelivered,
        SUM(COUNT(requestitemid)) OVER (
            PARTITION BY DATE(DATE_TRUNC('month', createdtime)), nhsnumberhash
        ) AS monthlyTotalMessages
    FROM request_item_status
    CROSS JOIN UNNEST(completedcommunicationtypes) AS t(communicationType)
    WHERE status = 'DELIVERED'
    GROUP BY 1, 2, 3, 4
)
WHERE monthlyTotalMessages >= 5
GROUP BY 1, 2, 3, 4
