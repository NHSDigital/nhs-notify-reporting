SELECT
    DATE(createdTime) AS createddate,
    DATE(DATE_TRUNC('month', createdtime)) AS createdMonth,
    nhsnumberhash,
    clientid,
    t.communicationType,
    requestitemid
FROM request_item_status
CROSS JOIN UNNEST(completedcommunicationtypes) AS t(communicationType)
WHERE status = 'DELIVERED'
